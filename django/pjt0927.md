버퍼를 활용하여 웹페이지에 렌더

```python

from django.shortcuts import render
from . models import Weather

import matplotlib.pyplot as plt

from io import BytesIO
import base64
import pandas as pd
# Create your views here.

def index(request):
    x=[1,2,3,4,5]
    y=[1,2,3,4,5]
    
    plt.clf()
    plt.plot(x,y)
    plt.title('Test Graph')
    plt.ylabel('y')
    plt.xlabel('x')
    
    buffer = BytesIO()

    plt.savefig(buffer,format='png')

    image_base64 = base64.b64encode(buffer.getvalue()).decode('utf-8').replace('\n','')
    buffer.close()
    context = {
        'chart_image' : f'data:image/png;base64,{image_base64}'
    }
    return render(request,'firsts/index.html',context)

def example(request):
    csv_path ='firsts/data/test_data.csv'
    df =pd.read_csv(csv_path)

    for row in df.iterrows():
        if Weather.objects.filter(date=row['Date']).exists():

            weather =Weather(
                date =row['Date'],
                temp_avg_f =row['TempAvgF'],
                events=row['Events'] if pd.notna(row['Events']) else ''
            )
            weather.save()

    weathers =Weather.objects.all()
    context ={
        'weathers' :weathers
    }
    return render(request,'firsts/example.html',context)
```
